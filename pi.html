import zipfile

face_cascade = cv.CascadeClassifier(cv.data.haarcascades + 'haarcascade_frontalface_default.xml')

def binarize(image_to_transform, threshold=255):
    output_image = image_to_transform.convert("L")
    for x in range(output_image.width):
        for y in range(output_image.height):
            if output_image.getpixel((x,y)) < threshold:
                output_image.putpixel((x,y),0)
            else:
                output_image.putpixel((x,y),255)
    return output_image

def erase_punctuation(str1):
    str2 = ""
    for character in str1:
        if character in string.ascii_lowercase:
            str2 = str2 + character
    return str2
    
def extract_words(img):
    image = Image.open("/home/luisrebollo/code/Python3/"+img)
    str1 = pytesseract.image_to_string(binarize(image))
    str1 = str1.lower()
    words_list = str1.split(" ")
    for word in range(len(words_list)):
        clean_word = erase_punctuation(words_list[word])
        words_list[word] = clean_word
    return words_list

def face_detector(img):
    cv_img = cv.imread("/home/luisrebollo/code/Python3/"+img)
    img_opcv = cv.cvtColor(cv_img,cv.COLOR_BGR2GRAY)
    (_, blackAndWhiteImage) = cv.threshold(cv_img, 120, 255, cv.THRESH_BINARY)
    blackAndWhiteImage = cv.convertScaleAbs(blackAndWhiteImage)
    faces = face_cascade.detectMultiScale(cv_img,1.35,4)
    faces_list = []
    pil_img = Image.open("/home/luisrebollo/code/Python3/"+img).convert("RGB")
    try:
        for x,y,w,h in faces:
            faces_list.append(pil_img.crop((x,y,x+w,y+h)))
        return faces_list
    except:
        return 0
